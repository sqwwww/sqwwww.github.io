用来 call coverage 和 snp 的流程，从reads到结果图

ref=ref/bowtie2Index/sub
threads=20
cat fq_list|while read id; do

arr=($id)
fq1=${arr[0]}
fq2=${arr[1]}
sample=`basename $fq1 _1.fq.gz`

echo ${sample}--------------------------
fastp -i $fq1\
-I $fq2\
-o fq/${sample}_clean_1.fastq.gz \
-O fq/${sample}_clean_2.fastq.gz \
--thread=$threads

bowtie2 -p ${threads} \
-x ${ref} \
-1 fq/${sample}_clean_1.fastq.gz \
-2 fq/${sample}_clean_2.fastq.gz \
-S ${sample}.sam

grep -P '@|AS:' ${sample}.sam | grep -v "XS:i" > ${sample}.uniq.sam
samtools view -@ 10 -bS  ${sample}.uniq.sam | samtools sort  -o ${sample}.uniq.bam
samtools index ${sample}.uniq.bam

done
